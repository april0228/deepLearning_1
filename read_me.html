原理

每一个像素作为一个特征点，一张照片的特征是特征点构成的特征向量X，对应一个权值向量W
z = wx + b
sigmoid = 1/(1+e^(-z))
loss使用交叉熵损失
计算dw,db w = w - rdw
b = b - r*db
反复使用上述神经网络前馈+反馈公式直到达到收敛效果即可

核心代码

def sigmoid(w,x,b):
    
    z = np.dot(x,w) + b
    a = 1/(1 + np.exp(-z))
    return a
    
def parameterInit() :
    w = np.zeros((12288,1))
    b = 0
    time = 20000
    r = 0.005
    return w,b,time,r

while(t <= time):

        loss = -np.sum(trainSetY*(np.log(A)) + (1-trainSetY)*(np.log(1-A))) / n  
        dz = A - trainSetY  
        dw = (np.dot(trainSetX.T,dz)) / n
        db = np.sum(dz) / n
        w = w - r*dw
        b = b - r*db
        if(t%100 == 0) :
            print(t,"th,loss = ",loss)
        t = t + 1
